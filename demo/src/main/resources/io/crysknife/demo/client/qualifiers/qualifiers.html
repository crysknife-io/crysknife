The @Qualifier annotation that is added to the annotation, makes this annotation discoverable by the container.

    }\n
    
    @Target({ElementType.TYPE, ElementType.METHOD, ElementType.PARAMETER, ElementType.FIELD})\n
    @Retention(RetentionPolicy.RUNTIME)\n
    @Documented\n
    @Qualifier\n
    public @interface QualifierOne {\n
    \n
    }\n
    
    @QualifierOne\n
    @Singleton\n
    public class QualifierBeanOne implements QualifierBean {\n
    \n

        }\n
    }\n
